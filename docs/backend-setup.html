/* ════════════════════════ HTTP entry ════════════════════════ */ function doGet(e) { try { const params = e.parameter || {}; const action = (params.action || 'tickets').toLowerCase(); const user = checkAuth_(params.email); } /* ════════════════════════ Sheet IO ════════════════════════ */ function ss_() { return SpreadsheetApp.getActiveSpreadsheet(); } function readSheet_(name, limit) { const sheet = ss_().getSheetByName(name);
